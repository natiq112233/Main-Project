    private void initializeSurveyPanel() {
        surveyPanel = new JPanel(new BorderLayout());
        questionPanel = new JPanel(new GridLayout(6, 1, 10, 10));
        progressBar = new JProgressBar(0, questions.size());
        progressBar.setValue(0);
        progressBar.setStringPainted(true);

        surveyPanel.add(progressBar, BorderLayout.NORTH);
        surveyPanel.add(questionPanel, BorderLayout.CENTER);
        surveyPanel.setBorder(BorderFactory.createEmptyBorder(20, 20, 20, 20));

        frame.add(surveyPanel);
        frame.revalidate();
        frame.repaint();
        displayNextQuestion();
    }

    private void displayNextQuestion() {
        questionPanel.removeAll();

        if (questionIndex < questions.size()) {
            String question = questions.get(questionIndex);
            JLabel questionLabel = new JLabel("<html><body style='width: 400px'>" + question + "</body></html>");
            questionLabel.setFont(new Font("Arial", Font.BOLD, 16));
            JTextField responseField = new JTextField();
            JButton nextButton = new JButton("Next");
            nextButton.setBackground(new Color(51, 153, 255));
            nextButton.setForeground(Color.WHITE);

            nextButton.addActionListener(new ActionListener() {
                @Override
                public void actionPerformed(ActionEvent e) {
                    String response = responseField.getText().trim();

                    if (response.isEmpty()) {
                        JOptionPane.showMessageDialog(frame, "Please fill in the answer before proceeding.", "Input Error", JOptionPane.WARNING_MESSAGE);
                    } else if (!isValidResponse(response, questionIndex)) {
                        JOptionPane.showMessageDialog(frame, "Invalid answer. Please enter 'yes', 'no', or a number between 1 and 5, as applicable.", "Input Error", JOptionPane.WARNING_MESSAGE);
                    } else {
                        responses.add(response);
                        progressBar.setValue(questionIndex + 1);
                        showTopicDetailsOption(question);
                        questionIndex++;
                        displayNextQuestion();
                    }
                }
            });

            questionPanel.add(questionLabel);
            questionPanel.add(responseField);
            questionPanel.add(nextButton);
        } else {
            frame.remove(surveyPanel);
            showSurveySummary();
        }

        frame.revalidate();
        frame.repaint();
    }

    private void showTopicDetailsOption(String question) {
        String topic = getTopicFromQuestion(question);

        if (topic != null && topicDetails.containsKey(topic)) {
            int result = JOptionPane.showConfirmDialog(frame, "Would you like to know more about this topic?", "More Information", JOptionPane.YES_NO_OPTION);
            if (result == JOptionPane.YES_OPTION) {
                showDetailedInfo(topic);
            }
        }
    }

    private String getTopicFromQuestion(String question) {
        String lowerQuestion = question.toLowerCase();
        if (lowerQuestion.contains("coding")) return "coding";
        if (lowerQuestion.contains("cybersecurity")) return "cybersecurity";
        if (lowerQuestion.contains("data")) return "data";
        if (lowerQuestion.contains("it systems")) return "it systems";
        if (lowerQuestion.contains("artificial intelligence")) return "artificial intelligence";
        if (lowerQuestion.contains("user interfaces")) return "user interfaces";
        if (lowerQuestion.contains("network administration")) return "network administration";
        if (lowerQuestion.contains("troubleshooting hardware")) return "hardware troubleshooting";
        if (lowerQuestion.contains("teaching technology")) return "teaching technology";
        return null;
    }

    private void showDetailedInfo(String topic) {
        JFrame detailFrame = new JFrame("Details about " + capitalize(topic));
        detailFrame.setSize(600, 450);
        detailFrame.setLayout(new BorderLayout());
        detailFrame.setLocationRelativeTo(frame);

        JTextArea detailsText = new JTextArea(topicDetails.get(topic));
        detailsText.setWrapStyleWord(true);
        detailsText.setLineWrap(true);
        detailsText.setEditable(false);
        detailsText.setFont(new Font("Serif", Font.PLAIN, 14));

        JScrollPane scrollPane = new JScrollPane(detailsText);
        detailFrame.add(scrollPane, BorderLayout.CENTER);

        JPanel buttonPanel = new JPanel();

        JButton pursueButton = new JButton("Pursue");
        pursueButton.setBackground(new Color(51, 153, 255));
        pursueButton.setForeground(Color.WHITE);
        pursueButton.addActionListener(e -> showPursueOptions(topic));

        JButton closeButton = new JButton("Back");
        closeButton.addActionListener(e -> detailFrame.dispose());

        buttonPanel.add(pursueButton);
        buttonPanel.add(closeButton);

        detailFrame.add(buttonPanel, BorderLayout.SOUTH);
        detailFrame.setVisible(true);
    }

    private void showPursueOptions(String topic) {
        JFrame pursueFrame = new JFrame("Pursue " + capitalize(topic));
        pursueFrame.setSize(500, 400);
        pursueFrame.setLayout(new BorderLayout());
        pursueFrame.setLocationRelativeTo(frame);

        JEditorPane pursuePane = new JEditorPane();
        pursuePane.setContentType("text/html");
        pursuePane.setEditable(false);

        String resources = getPursueResources(topic, "");
        pursuePane.setText(resources);

        pursuePane.addHyperlinkListener(e -> {
            if (e.getEventType() == HyperlinkEvent.EventType.ACTIVATED) {
                String url = e.getDescription();

                if (url.startsWith("http")) {
                    try {
                        Desktop.getDesktop().browse(new URI(url));
                    } catch (Exception ex) {
                        ex.printStackTrace();
                    }
                } else {
                    String subtopic = parseSubtopicFromURL(url); // Extract subtopic
                    String updatedResources = getPursueResources(topic, subtopic); // Load subtopic content
                    pursuePane.setText(updatedResources);
                }
            }
        });

        JScrollPane scrollPane = new JScrollPane(pursuePane);
        pursueFrame.add(scrollPane, BorderLayout.CENTER);

        JButton closeButton = new JButton("Back");
        closeButton.addActionListener(e -> pursueFrame.dispose());
        pursueFrame.add(closeButton, BorderLayout.SOUTH);

        pursueFrame.setVisible(true);
    }

    private String parseSubtopicFromURL(String url) {
        String[] parts = url.split("=");
        return parts.length > 1 ? parts[1] : "";
    }

    private void showSurveySummary() {
        JPanel summaryPanel = new JPanel(new GridLayout(questions.size() + 1, 1, 5, 5));
        summaryPanel.setBorder(BorderFactory.createEmptyBorder(10, 10, 10, 10));
        summaryPanel.setBackground(Color.WHITE);

        JLabel summaryLabel = new JLabel("Survey Summary:");
        summaryLabel.setFont(new Font("Arial", Font.BOLD, 18));
        summaryLabel.setForeground(new Color(0, 102, 204));
        summaryPanel.add(summaryLabel);

        for (int i = 0; i < questions.size(); i++) {
            summaryPanel.add(new JLabel((i + 1) + ". " + questions.get(i) + ": " + responses.get(i)));
        }

        int result = JOptionPane.showConfirmDialog(frame, summaryPanel, "Survey Summary", JOptionPane.OK_CANCEL_OPTION, JOptionPane.PLAIN_MESSAGE);
        if (result == JOptionPane.OK_OPTION) {
            initializeResultsPanel();
        } else {
            JOptionPane.showMessageDialog(frame, "You can retake the survey anytime.", "Survey Incomplete", JOptionPane.INFORMATION_MESSAGE);
            frame.dispose();
        }
    }

    private void initializeResultsPanel() {
        resultsPanel = new JPanel(new BorderLayout());
        resultsPanel.setBorder(BorderFactory.createEmptyBorder(20, 20, 20, 20));
        resultsPanel.setBackground(new Color(230, 240, 255));

        JLabel suggestionLabel = new JLabel("Career Suggestions:");
        suggestionLabel.setFont(new Font("Arial", Font.BOLD, 18));
        suggestionLabel.setForeground(new Color(0, 102, 204));
        resultsPanel.add(suggestionLabel, BorderLayout.NORTH);

        JTextArea suggestionsText = new JTextArea();
        suggestionsText.setEditable(false);
        suggestionsText.setFont(new Font("Arial", Font.PLAIN, 14));
        suggestionsText.setLineWrap(true);
        suggestionsText.setWrapStyleWord(true);
        suggestionsText.setText(generateCareerSuggestions());

        JScrollPane scrollPane = new JScrollPane(suggestionsText);
        resultsPanel.add(scrollPane, BorderLayout.CENTER);

        JButton feedbackButton = new JButton("Close");
        feedbackButton.setBackground(new Color(51, 153, 255));
        feedbackButton.setForeground(Color.WHITE);
        feedbackButton.addActionListener(e -> {
            JOptionPane.showMessageDialog(frame, "Thank you for using the Career Interest Matcher!", "Goodbye", JOptionPane.INFORMATION_MESSAGE);
            frame.dispose();
        });

        JPanel buttonPanel = new JPanel();
        buttonPanel.add(feedbackButton);
        resultsPanel.add(buttonPanel, BorderLayout.SOUTH);

        frame.add(resultsPanel);
        frame.revalidate();
        frame.repaint();
    }
    private String generateCareerSuggestions() {
        StringBuilder suggestions = new StringBuilder();

        if (responses.contains("yes_programming") && getRating("Rate your programming skills (1-5)") >= 2) {
            suggestions.append("- Software Developer\n");
            suggestions.append("- Web Developer\n");
            suggestions.append("- Game Programmer\n\n");
        }

        if (responses.contains("yes_cybersecurity") && getRating("Rate your analytical skills (1-5)") >= 3) {
            suggestions.append("- Cybersecurity Analyst\n");
            suggestions.append("- Security Consultant\n");
            suggestions.append("- Penetration Tester\n\n");
        }

        if (responses.contains("yes_data") && getRating("Rate your analytical skills (1-5)") >= 4) {
            suggestions.append("- Data Scientist\n");
            suggestions.append("- Data Analyst\n");
            suggestions.append("- Data Engineer\n\n");
        }

        if (suggestions.length() == 0) {
            suggestions.append("Consider exploring various fields to find the best fit for your interests and skills.");
        }

        return suggestions.toString();
    }

    private int getRating(String question) {
        for (int i = 0; i < questions.size(); i++) {
            if (questions.get(i).equals(question)) {
                try {
                    return Integer.parseInt(responses.get(i));
                } catch (NumberFormatException e) {
                    return 0;
                }
            }
        }
        return 0;
    }

    private boolean isValidSignUp(String username, String password, String email) {
        return password.matches("^(?=.*[A-Za-z])(?=.*\\d)[A-Za-z\\d]{4,}$") && email.endsWith("@gmail.com");
    }

    private void saveUserData(String username, String password, String email) {
        try (BufferedWriter writer = new BufferedWriter(new FileWriter("user_data.txt", true))) {
            writer.write("Username: " + username);
            writer.newLine();
            writer.write("Password: " + password);
            writer.newLine();
            writer.write("Email: " + email);
            writer.newLine();
            writer.write("-----");
            writer.newLine();
        } catch (IOException ex) {
            JOptionPane.showMessageDialog(frame, "Error saving user data.", "File Error", JOptionPane.ERROR_MESSAGE);
        }
    }

    private boolean validateLogin(String username, String password) {
        File file = new File("user_data.txt");
        if (!file.exists()) {
            JOptionPane.showMessageDialog(frame, "User data file not found.", "File Error", JOptionPane.ERROR_MESSAGE);
            return false;
        }

        try (Scanner scanner = new Scanner(file)) {
            while (scanner.hasNextLine()) {
                String fileUsernameLine = scanner.nextLine();
                String filePasswordLine = scanner.nextLine();
                scanner.nextLine();
                if (scanner.hasNextLine()) scanner.nextLine();

                String[] usernameParts = fileUsernameLine.split(": ");
                String[] passwordParts = filePasswordLine.split(": ");

                if (usernameParts.length < 2 || passwordParts.length < 2) continue;

                String fileUsername = usernameParts[1];
                String filePassword = passwordParts[1];

                if (fileUsername.equals(username) && filePassword.equals(password)) {
                    return true;
                }
            }
        } catch (FileNotFoundException e) {
            JOptionPane.showMessageDialog(frame, "User data file not found.", "File Error", JOptionPane.ERROR_MESSAGE);
        }
        return false;
    }

    private boolean isValidResponse(String response, int questionIndex) {
        if (questionIndex < 10) {
            return response.equalsIgnoreCase("yes") || response.equalsIgnoreCase("no");
        } else {
            try {
                int rating = Integer.parseInt(response);
                return rating >= 1 && rating <= 5;
            } catch (NumberFormatException e) {
                return false;
            }
        }
    }

    private String capitalize(String word) {
        return word.substring(0, 1).toUpperCase() + word.substring(1);
    }
    private void openLink(String url) {
        try {
            URI uri = new URI(url);
            Desktop.getDesktop().browse(uri);
        } catch (Exception e) {
            JOptionPane.showMessageDialog(frame, "Error opening the link: " + url, "Link Error", JOptionPane.ERROR_MESSAGE);
        }
    }
